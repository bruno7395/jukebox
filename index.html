Mise à jour de la fonction qui scanne les tags et récupère l'image async function processTags(i) { if (i >= tracks.length) { document.getElementById('status').innerText = "Scan terminé."; return; } window.jsmediatags.read(tracks[i].file, { onSuccess: (t) => { tracks[i].title = t.tags.title || tracks[i].title; tracks[i].artist = t.tags.artist || "Inconnu"; tracks[i].album = t.tags.album || "Album Inconnu"; // Si on trouve une pochette dans le fichier, on la mémorise if (t.tags.picture) { const base64 = "data:" + t.tags.picture.format + ";base64," + btoa(String.fromCharCode(...t.tags.picture.data)); albumCovers[tracks[i].album] = base64; localStorage.setItem('covers', JSON.stringify(albumCovers)); } render(); processTags(i + 1); }, onError: () => processTags(i + 1) }); } // 2. Mise à jour du rendu pour afficher l'image mémorisée function render() { document.getElementById('music-list-body').innerHTML = tracks.map((t, i) => ` ${t.title} ${t.artist} `).join(''); const albums = [...new Set(tracks.map(t => t.album))]; document.getElementById('view-albums').innerHTML = albums.map(a => `
${albumCovers[a] ? `` : ``}

${a}

`).join(''); }
